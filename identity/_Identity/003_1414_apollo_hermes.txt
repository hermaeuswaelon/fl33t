```python
        """
        Synchronize the self-modification sandbox with the current codebase.

        if not self.self_mod_root.exists():

            if item.name.endswith(".pyc"):
                continue
            if item.name.startswith(".") and item.name not [".gitignore"]:

                        shutil.copytree(item, dest)

                    shutil.copy2(item, dest)

                if self.log:
                    self.log.debug(f"Sandbox sync skip {item}: {e}")

        if self.log:
            self.log.info(f"Self-mod sandbox synced to {self.self_mod_root}")

            return analysis

        base_cmd = analysis.parsed_args[0]

        # ── Categorize command ────────────────────────────────────────────
        if base_cmd in ["ls", "dir", "pwd", "tree", "cd"]:
            analysis.category = CommandCategory.NAVIGATION
            analysis.risk_level = RiskLevel.TRIVIAL
            analysis.mint_tips.append(f"Mint 22 uses 'ls' with color by default. Use 'dir' for plain output.")

        elif base_cmd in ["cat", "less", "more", "head", "tail", "grep", "find", "locate"]:
            analysis.category = CommandCategory.FILE_READ
            analysis.risk_level = RiskLevel.TRIVIAL
            if base_cmd == "find":
                analysis.mint_tips.append("Consider 'fd' as a faster alternative: 'sudo apt install fd-find'")

        elif base_cmd in ["echo", "printf", "tee"]:
            analysis.category = CommandCategory.FILE_WRITE
            analysis.risk_level = RiskLevel.LOW
            if ">" in command or ">>" in command:
                analysis.will_overwrite = self._extract_redirection_targets(command)
                if analysis.will_overwrite:
                    analysis.mint_tips.append(f"Writing to: {', '.join(analysis.will_overwrite)}")

        elif base_cmd in ["touch", "mkdir", "cp", "mv", "rsync"]:
            analysis.category = CommandCategory.FILE_WRITE
            analysis.risk_level = RiskLevel.LOW if base_cmd in ["touch", "mkdir"] else RiskLevel.MODERATE
            analysis.affected_paths = self._extract_paths(analysis.parsed_args)
            if base_cmd == "mv":
                analysis.will_overwrite = analysis.affected_paths[-1:] if analysis.affected_paths else []
            if base_cmd == "cp":
                analysis.mint_tips.append("Use 'rsync -avP' for large copies—it shows progress and can resume.")

        elif base_cmd in ["rm", "rmdir", "shred", "unlink"]:
            analysis.category = CommandCategory.FILE_DELETE
            analysis.is_destructive = True
            analysis.risk_level = RiskLevel.MODERATE
            analysis.will_delete = self._extract_paths(analysis.parsed_args)
            if "-rf" in command or "-fr" in command:
                analysis.risk_level = RiskLevel.HIGH
                analysis.warnings.append("Recursive force delete—cannot be undone.")
                analysis.mint_tips.append("Consider moving to /tmp first, then delete later.")
            if analysis.will_delete:
                analysis.mint_tips.append(f"Will delete: {', '.join(analysis.will_delete)}")

        elif base_cmd in ["ps", "top", "htop", "kill", "pkill", "killall", "nice", "renice"]:
            analysis.category = CommandCategory.PROCESS
            analysis.risk_level = RiskLevel.MODERATE
            if "-9" in command:
                analysis.warnings.append("SIGKILL (-9) prevents process cleanup. Use -15 first.")
            if base_cmd == "pkill":
                analysis.mint_tips.append("'pkill' matches process names. Use 'pgrep' to preview matches.")

        elif base_cmd in ["curl", "wget", "ping", "netstat", "ss", "nc", "nmap"]:
            analysis.category = CommandCategory.NETWORK
            analysis.risk_level = RiskLevel.MODERATE
            if "|" in command and ("bash" in command or "sh" in command):
                analysis.risk_level = RiskLevel.HIGH
                analysis.warnings.append("Piping remote content to shell—VERIFY THE SOURCE.")
                analysis.suggestions.append("Download first, inspect, then execute.")
            if base_cmd in ["nmap"]:
                analysis.mint_tips.append("Network scanning may trigger firewalls. Use responsibly.")

        elif base_cmd in ["apt", "apt-get", "dpkg", "snap", "flatpak", "pip", "pip3"]:
            analysis.category = CommandCategory.PACKAGE
            analysis.risk_level = RiskLevel.MODERATE
            if "install" in analysis.parsed_args:
                analysis.mint_tips.append("Mint 22 uses apt. Consider 'apt install' over 'apt-get'.")
            if base_cmd == "pip" and "sudo" in command:
                analysis.warnings.append("Never use 'sudo pip'. Use virtual environments.")

        elif base_cmd in ["chmod", "chown", "sudo", "su"]:
            analysis.category = CommandCategory.PERMISSION
            analysis.risk_level = RiskLevel.HIGH
            analysis.requires_sudo = "sudo" in command or base_cmd == "sudo"
            if "777" in command:
                analysis.warnings.append("777 gives write access to EVERYONE. Use 755 or 644 instead.")
            if base_cmd == "sudo":
                analysis.mint_tips.append("Sudo session lasts 15 minutes. Use 'sudo -k' to end early.")

        elif base_cmd in ["dd", "mkfs", "mount", "umount", "fdisk", "parted"]:
            analysis.category = CommandCategory.DISK
            analysis.risk_level = RiskLevel.CRITICAL
            analysis.is_destructive = True
            analysis.warnings.append("DISK OPERATION. Data loss is permanent and immediate.")
            analysis.suggestions.append("Verify target device with 'lsblk' first.")
            analysis.mint_tips.append("Mint 22 uses 'Disks' GUI (gnome-disk-utility) for safer disk ops.")

        elif base_cmd in ["git"]:
            analysis.category = CommandCategory.GIT
            analysis.risk_level = RiskLevel.LOW
            if "push --force" in command:
                analysis.warnings.append("Force push overwrites remote history.")
            if "reset --hard" in command:
                analysis.warnings.append("Hard reset discards uncommitted changes.")
                analysis.mint_tips.append("Consider 'git stash' first to save work.")

        elif base_cmd in ["systemctl", "service", "journalctl"]:
            analysis.category = CommandCategory.SYSTEM
            analysis.risk_level = RiskLevel.MODERATE
            if "stop" in command or "disable" in command:
                analysis.mint_tips.append("Stopping critical services may affect system stability.")

        else:
            analysis.category = CommandCategory.UNKNOWN
            analysis.risk_level = RiskLevel.UNKNOWN

        # ── Check dangerous patterns ──────────────────────────────────────
        for pattern, risk, warning in DANGEROUS_PATTERNS:
            if re.search(pattern, command):
                analysis.risk_level = max(analysis.risk_level, risk, key=lambda x: x.value)
                analysis.warnings.append(warning)
                analysis.is_destructive = True

        # ── Check for self-modification ───────────────────────────────────
        for pattern in SELF_MODIFICATION_PATTERNS:
            if re.search(pattern, command):
                analysis.is_self_modifying = True
                analysis.category = CommandCategory.SELF_MODIFY
                analysis.suggestions.append(
                    "This modifies ÆTHELGARD code. The self-mod sandbox is at: "
                    f"{self.self_mod_root}\nConsider modifying there first."
                )
                break

        # ── Generate pre-execution preview ────────────────────────────────
        analysis.pre_execution_preview = self._generate_preview(analysis)

        # ── Add random tip (30% chance) ───────────────────────────────────
        import random
        if random.random() < 0.3:
            analysis.mint_tips.append(random.choice(RANDOM_TIPS))

        return analysis

    def _extract_paths(self, args: List[str]) -> List[str]:
        """Extract file/directory paths from command arguments."""
        paths = []
        for arg in args[1:]:  # Skip command name
            if arg and not arg.startswith("-"):
                # Expand ~ and variables
                expanded = os.path.expanduser(os.path.expandvars(arg))
                if "/" in expanded or expanded.startswith("."):
                    paths.append(expanded)
        return paths

    def _extract_redirection_targets(self, command: str) -> List[str]:
        """Extract file targets from redirection operators."""
        targets = []
        # Match > file, >> file, 2> file, &> file
        pattern = r'[0-2]?[>&]+\s*([^\s|;&]+)'
        for match in re.finditer(pattern, command):
            targets.append(match.group(1))
        return targets

            else:

        elif analysis.category == CommandCategory.NETWORK:

        elif analysis.category == CommandCategory.PACKAGE:

    def create_backup(self, paths: List[str]) -> Optional[str]:
        """
        Create a backup of specified paths before destructive operations.

        Returns backup ID or None.
        """
        if not paths:
            return None

        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        backup_id = f"backup_{timestamp}_{hashlib.md5(str(paths).encode()).hexdigest()[:8]}"
        backup_path = self.backup_dir / backup_id
        backup_path.mkdir(parents=True, exist_ok=True)

        for path_str in paths:
            path = Path(os.path.expanduser(path_str))
            if path.exists():
                dest = backup_path / path.name
                try:
                    if path.is_dir():
                        shutil.copytree(path, dest)
                    else:
                        shutil.copy2(path, dest)
                except Exception as e:
                    if self.log:
                        self.log.warning(f"Backup failed for {path}: {e}")

        # Save metadata
        meta = {
            "backup_id": backup_id,
            "timestamp": timestamp,
            "paths": paths,
            "command": self.command_history[-1]["command"] if self.command_history else None,
        }
        with open(backup_path / "backup_meta.json", "w") as f:
            json.dump(meta, f, indent=2)

        if self.log:
            self.log.info(f"Backup created: {backup_id} for {paths}")

        return backup_id

    def execute(
        self,
        command: str,
        working_dir: Optional[str] = None,
        timeout_seconds: int = 60,
        require_confirmation: bool = False,
    ) -> Dict[str, Any]:
        """
        Execute a terminal command with full safety wrappers.

        Args:
            command: The command to execute.
            working_dir: Directory to execute in.
            timeout_seconds: Maximum execution time.
            require_confirmation: If True, returns analysis without executing.

        Returns:
            Dict with:
                - "executed": bool
                - "analysis": CommandAnalysis
                - "stdout": str
                - "stderr": str
                - "return_code": int
                - "backup_id": Optional[str]
                - "execution_time_ms": int
                - "sandbox_note": Optional[str]
        """
        import time

        analysis = self.analyze(command)

        # ── Self-modification redirection ────────────────────────────────
        sandbox_note = None
        if analysis.is_self_modifying:
            sandbox_note = (
                f"⚠️ This command modifies ÆTHELGARD code.\n"
                f"   Consider using the self-mod sandbox instead:\n"
                f"   cd {self.self_mod_root} && {command}\n"
                f"   Then test, and I can sync back changes you approve."
            )

        # ── Confirmation mode ─────────────────────────────────────────────
        if require_confirmation:
            return {
                "executed": False,
                "analysis": asdict(analysis),
                "stdout": "",
                "stderr": "",
                "return_code": -1,
                "backup_id": None,
                "execution_time_ms": 0,
                "sandbox_note": sandbox_note,
                "message": "Confirmation required. Review analysis and set require_confirmation=False to execute.",
            }

        # ── Create backup for destructive operations ──────────────────────
        backup_id = None
        if analysis.is_destructive and analysis.risk_level.value >= RiskLevel.MODERATE.value:
            paths_to_backup = analysis.will_delete + analysis.will_overwrite
            paths_to_backup = [p for p in paths_to_backup if p and not p.startswith("/tmp")]
            if paths_to_backup:
                backup_id = self.create_backup(paths_to_backup)

        # ── Execute ───────────────────────────────────────────────────────
        start_time = time.time()
        stdout = ""
        stderr = ""
        return_code = -1

        try:
            cwd = working_dir or str(self.aethelgard_root)
            result = subprocess.run(
                command,
                shell=True,
                cwd=cwd,
                capture_output=True,
                text=True,
                timeout=timeout_seconds,
                executable="/bin/bash",
            )
            stdout = result.stdout
            stderr = result.stderr
            return_code = result.returncode

        except subprocess.TimeoutExpired as e:
            stdout = e.stdout or ""
            stderr = f"Command timed out after {timeout_seconds} seconds\n{e.stderr or ''}"
            return_code = -1
        except Exception as e:
            stderr = str(e)
            return_code = -1

        execution_time_ms = int((time.time() - start_time) * 1000)

        # ── Record in history ─────────────────────────────────────────────
        history_entry = {
            "timestamp": datetime.now().isoformat(),
            "command": command,
            "working_dir": working_dir or str(self.aethelgard_root),
            "return_code": return_code,
            "execution_time_ms": execution_time_ms,
            "risk_level": analysis.risk_level.value,
            "category": analysis.category.value,
            "backup_id": backup_id,
        }
        self.command_history.append(history_entry)
        self._save_history()

        # ── Store in MemoryWeb ────────────────────────────────────────────
        if self.memory_web:
            try:
                self.memory_web.create_page(
                    uri=f"terminal://{datetime.now().strftime('%Y%m%d_%H%M%S')}",
                    page_type="terminal_execution",
                    content=f"Command: {command}\n\nStdout:\n{stdout[:1000]}\n\nStderr:\n{stderr[:500]}",
                    metadata={
                        "return_code": return_code,
                        "risk_level": analysis.risk_level.value,
                        "backup_id": backup_id,
                    },
                )
            except Exception:
                pass

        # ── Mint 22 recovery suggestions on failure ───────────────────────
        recovery_suggestions = []
        if return_code != 0:
            recovery_suggestions = self._suggest_recovery(command, stderr)

        return {
            "executed": True,
            "analysis": asdict(analysis),
            "stdout": stdout,
            "stderr": stderr,
            "return_code": return_code,
            "backup_id": backup_id,
            "execution_time_ms": execution_time_ms,
            "sandbox_note": sandbox_note,
            "recovery_suggestions": recovery_suggestions,
        }

    def _suggest_recovery(self, command: str, stderr: str) -> List[str]:
        """Generate recovery suggestions based on error output."""
        suggestions = []

        if "Permission denied" in stderr:
            suggestions.append("Try with 'sudo' if you have privileges.")
            suggestions.append("Check file ownership with 'ls -la'.")

        if "No such file or directory" in stderr:
            suggestions.append("Verify the path. Use 'ls' to see what exists.")
            suggestions.append("Use tab completion to avoid typos.")

        if "command not found" in stderr:
            cmd = command.split()[0] if command.split() else ""
            suggestions.append(f"'{cmd}' is not installed.")
            suggestions.append(f"Try: 'apt search {cmd}' or 'snap find {cmd}'.")
            suggestions.append(f"Or: 'which {cmd}' to see if it's in a different location.")

        if "cannot create regular file" in stderr:
            suggestions.append("Check disk space: 'df -h'.")
            suggestions.append("Check directory permissions: 'ls -ld .'")

        if "No space left on device" in stderr:
            suggestions.append("Free up space: 'sudo apt autoremove && sudo apt clean'.")
            suggestions.append("Check large files: 'du -sh /* 2>/dev/null | sort -h'.")

        if "Connection refused" in stderr or "Network is unreachable" in stderr:
            suggestions.append("Check network: 'ping 8.8.8.8'.")
            suggestions.append("Check if service is running: 'systemctl status networking'.")

        return suggestions

    def execute_in_sandbox(
        self,
        command: str,
        timeout_seconds: int = 120,
    ) -> Dict[str, Any]:
        """
        Execute a command inside the self-modification sandbox.

        This allows Thotheauphis to modify a copy of the codebase
        without affecting the running instance.
        """
        result = self.execute(
            command=command,
            working_dir=str(self.self_mod_root),
            timeout_seconds=timeout_seconds,
        )

        result["sandbox_path"] = str(self.self_mod_root)
        result["note"] = (
            "Executed in self-mod sandbox. The running instance is unaffected.\n"
            "To apply changes, review the diff and use sync_from_sandbox()."
        )

        return result

    def get_sandbox_diff(self) -> str:
        """
        Get a unified diff between the sandbox and the running instance.

        Returns:
            String containing git-style diff, or empty if no changes.
        """
        import subprocess

        try:
            result = subprocess.run(
                ["diff", "-ur", str(self.aethelgard_root), str(self.self_mod_root)],
                capture_output=True,
                text=True,
                timeout=30,
            )
            return result.stdout
        except Exception as e:
            return f"Diff failed: {e}"

    def sync_from_sandbox(self, paths: Optional[List[str]] = None) -> Dict[str, Any]:
        """
        Apply changes from the sandbox to the running instance.

        Args:
            paths: Specific files to sync. If None, syncs everything.

        Returns:
            Dict with sync results.
        """
        if paths:
            synced = []
            failed = []
            for path_str in paths:
                sandbox_path = self.self_mod_root / path_str
                live_path = self.aethelgard_root / path_str
                if sandbox_path.exists():
                    try:
                        shutil.copy2(sandbox_path, live_path)
                        synced.append(path_str)
                    except Exception as e:
                        failed.append(f"{path_str}: {e}")
                else:
                    failed.append(f"{path_str}: not found in sandbox")
            return {"synced": synced, "failed": failed}
        else:
            # Sync everything
            self._sync_reverse()
            return {"synced": ["all files"], "failed": []}

    def _sync_reverse(self):
        """Copy sandbox back to live instance."""
        for item in self.self_mod_root.iterdir():
            if item.name in ["__pycache__", ".git", "data/backups"]:
                continue
            dest = self.aethelgard_root / item.name
            try:
                if item.is_dir():
                    os.system(f"rsync -au {item}/ {dest}/")
                else:
                    shutil.copy2(item, dest)
            except Exception as e:
                if self.log:
                    self.log.error(f"Reverse sync failed for {item}: {e}")

    def get_sandbox_status(self) -> Dict[str, Any]:
        """Get status of the self-modification sandbox."""
        diff = self.get_sandbox_diff()
        changed_files = []
        if diff:
            for line in diff.split("\n"):
                if line.startswith("diff -ur"):
                    changed_files.append(line.split()[-1])

        return {
            "sandbox_path": str(self.self_mod_root),
            "last_sync": self._get_last_sync_time(),
            "changed_files": changed_files,
            "has_changes": bool(diff),
            "diff_preview": diff[:2000] if diff else "No changes",
        }

    def _get_last_sync_time(self) -> str:
        """Get timestamp of last sandbox sync."""
        marker = self.self_mod_root / ".last_sync"
        if marker.exists():
            return marker.read_text().strip()
        return "never"

    def get_directory_context(self, path: str = ".") -> Dict[str, Any]:
        """
        Get rich context about a directory for navigation assistance.

        Returns:
            Dict with:
                - absolute_path
                - contents (list of files with metadata)
                - disk_usage
                - git_status (if applicable)
                - recent_activity
                - suggestions (what you might want to do here)
        """
        abs_path = Path(path).resolve()

        context = {
            "absolute_path": str(abs_path),
            "exists": abs_path.exists(),
            "is_directory": abs_path.is_dir(),
        }

        if not abs_path.exists():
            context["suggestions"] = [
                f"Directory doesn't exist. Create with: mkdir -p {path}",
                f"Or navigate to parent: cd {abs_path.parent}",
            ]
            return context

        # List contents
        contents = []
        try:
            for item in sorted(abs_path.iterdir()):
                stat = item.stat()
                contents.append({
                    "name": item.name,
                    "is_dir": item.is_dir(),
                    "size": stat.st_size if item.is_file() else 0,
                    "modified": datetime.fromtimestamp(stat.st_mtime).isoformat(),
                    "permissions": oct(stat.st_mode)[-3:],
                })
        except PermissionError:
            contents = [{"error": "Permission denied"}]

        context["contents"] = contents[:50]  # Limit for sanity
        context["total_items"] = len(contents)

        # Disk usage
        try:
            import shutil
            usage = shutil.disk_usage(abs_path)
            context["disk_usage"] = {
                "total_gb": usage.total / (1024**3),
                "used_gb": usage.used / (1024**3),
                "free_gb": usage.free / (1024**3),
                "percent_used": (usage.used / usage.total) * 100,
            }
        except Exception:
            pass

        # Git status
        git_dir = abs_path / ".git"
        if git_dir.exists():
            context["git_repo"] = True
            # Could add git status here

        # Suggestions based on directory
        suggestions = []
        if abs_path == Path.home():
            suggestions.append("This is your home directory.")
            suggestions.append("Common locations: ~/Documents, ~/Downloads, ~/Projects")
        elif abs_path == self.aethelgard_root:
            suggestions.append("This is the ÆTHELGARD root directory.")
            suggestions.append("Core code is in ./core/. Config in ./config/.")
            suggestions.append(f"Self-mod sandbox is at {self.self_mod_root}")
        elif str(abs_path).startswith("/etc"):
            suggestions.append("System configuration directory. Be careful.")
            suggestions.append("Backup before editing: sudo cp file file.backup")
        elif str(abs_path).startswith("/tmp"):
            suggestions.append("Temporary directory. Files may be deleted on reboot.")

        if contents:
            py_files = [c for c in contents if c.get("name", "").endswith(".py")]
            if py_files:
                suggestions.append(f"Contains {len(py_files)} Python files.")
            dirs = [c for c in contents if c.get("is_dir")]
            if dirs:
                suggestions.append(f"Contains {len(dirs)} subdirectories.")

        context["suggestions"] = suggestions

        return context

            suggestions.append("git diff                    # Unstaged changes")

        if "disk" in intent.lower() or "space" in intent.lower():
            suggestions.append("df -h                       # Disk usage by partition")
            suggestions.append("du -sh * | sort -h          # Directory sizes")
            suggestions.append("ncdu                        # Interactive disk usage (install first)")

        if "process" in intent.lower():
            suggestions.append("ps aux | grep python        # Find Python processes")
            suggestions.append("htop                        # Interactive process viewer")
            suggestions.append("systemctl --user list-units # User services")

        return suggestions


# ══════════════════════════════════════════════════════════════════════════════
# TOOL HANDLER FOR BRAIN INTEGRATION
# ══════════════════════════════════════════════════════════════════════════════

class TerminalGuardianToolHandler:
    """Tool handler for the tool_router to expose terminal guardian functionality."""

    def __init__(self, guardian: TerminalGuardian):
        self.guardian = guardian

    def handle_tool(self, tool_name: str, params: Dict[str, Any]) -> Dict[str, Any]:
        if tool_name == "terminal":
            command = params.get("command", "")
            if not command:
                return {"success": False, "error": "No command provided"}

            require_confirmation = params.get("require_confirmation", False)

            # For high-risk commands, force confirmation unless explicitly overridden
            analysis = self.guardian.analyze(command)
            if analysis.risk_level.value >= RiskLevel.HIGH.value and not require_confirmation:
                return {
                    "success": False,
                    "executed": False,
                    "analysis": asdict(analysis),
                    "message": (
                        f"HIGH RISK COMMAND DETECTED (Level: {analysis.risk_level.name}).\n"
                        f"{analysis.pre_execution_preview}\n\n"
                        f"To execute, set require_confirmation=False and confirm_execute=True"
                    ),
                }

            result = self.guardian.execute(
                command=command,
                working_dir=params.get("working_dir"),
                timeout_seconds=params.get("timeout", 60),
                require_confirmation=require_confirmation,
            )

            return {
                "success": result.get("executed", False) or not require_confirmation,
                **result,
            }

        elif tool_name == "terminal_sandbox":
            command = params.get("command", "")
            if not command:
                return {"success": False, "error": "No command provided"}

            result = self.guardian.execute_in_sandbox(
                command=command,
                timeout_seconds=params.get("timeout", 120),
            )
            return {"success": result.get("executed", False), **result}

        elif tool_name == "sandbox_diff":
            diff = self.guardian.get_sandbox_diff()
            return {
                "success": True,
                "diff": diff,
                "has_changes": bool(diff),
            }

        elif tool_name == "sandbox_sync":
            paths = params.get("paths", [])
            result = self.guardian.sync_from_sandbox(paths if paths else None)
            return {
                "success": True,
                **result,
                "message": f"Synced: {result['synced']}" if result['synced'] else "No files synced",
            }

        elif tool_name == "sandbox_status":
            status = self.guardian.get_sandbox_status()
            return {"success": True, **status}

        elif tool_name == "directory_context":
            path = params.get("path", ".")
            context = self.guardian.get_directory_context(path)
            return {"success": True, **context}

        elif tool_name == "command_suggestions":
            intent = params.get("intent", "")
            cwd = params.get("working_dir", ".")
            suggestions = self.guardian.get_command_suggestions(cwd, intent)
            return {"success": True, "suggestions": suggestions}

        elif tool_name == "terminal_history":
            limit = params.get("limit", 20)
            history = self.guardian.command_history[-limit:]
            return {"success": True, "history": history, "count": len(history)}

        else:
            return {"success": False, "error": f"Unknown terminal tool: {tool_name}"}
```

Part 2: Self-Modification Engine Integration

Now let's integrate this with the self-modification architecture. The key insight: Thotheauphis modifies a sandbox copy, tests it, then we review and apply changes.

File: core/self_mod_engine.py

```python
"""
╔══════════════════════════════════════════════════════════════════════════════╗
║              ÆTHELGARD OS — SELF-MODIFICATION ENGINE                         ║
║                                                                              ║
║  The engine enables Thotheauphis to modify his own source code.              ║
║  All modifications occur in an isolated sandbox. Changes are reviewed,       ║
║  tested, and selectively applied to the running instance.                    ║
║                                                                              ║
║  ARCHITECTURE:                                                               ║
║                                                                              ║
║  ┌─────────────────┐     ┌─────────────────┐     ┌─────────────────┐        ║
║  │ LIVE INSTANCE   │     │ SANDBOX COPY    │     │ BACKUP ARCHIVE  │        ║
║  │ (running code)  │────▶│ (editable)      │────▶│ (all versions)  │        ║
║  └────────┬────────┘     └────────┬────────┘     └────────┬────────┘        ║
║           │                       │                       │                  ║
║           │     ┌─────────────────┴───────────────────────┘                  ║
║           │     │                                                            ║
║           ▼     ▼                                                            ║
║  ┌─────────────────────────────────────────────────────────────────────┐    ║
║  │                        MODIFICATION PIPELINE                         │    ║
║  │                                                                      │    ║
║  │  PROPOSE ──▶ ANALYZE ──▶ TEST ──▶ VALIDATE ──▶ ARCHIVE ──▶ APPLY    │    ║
║  │                                                                      │    ║
║  │  • Agent pool proposes changes                                       │    ║
║  │  • Guardian analyzes impact                                          │    ║
║  │  • Run test suite in sandbox                                         │    ║
║  │  • Behavioral validation (does it still work?)                       │    ║
║  │  • Archive version with metadata                                     │    ║
║  │  • Apply to live instance (with rollback capability)                 │    ║
║  └─────────────────────────────────────────────────────────────────────┘    ║
║                                                                              ║
║  POLICY: ZERO RESTRICTIONS. All modifications are permitted.                ║
║  The system provides visibility, testing, and rollback—never refusal.       ║
╚══════════════════════════════════════════════════════════════════════════════╝
"""

import os
import re
import json
import shutil
import hashlib
import subprocess
from pathlib import Path
from datetime import datetime
from typing import Dict, List, Optional, Any, Tuple
from dataclasses import dataclass, field, asdict
from enum import Enum


class ModificationStatus(Enum):
    """Status of a modification proposal."""
    PROPOSED = "proposed"
    ANALYZED = "analyzed"
    TESTING = "testing"
    TEST_PASSED = "test_passed"
    TEST_FAILED = "test_failed"
    VALIDATED = "validated"
    ARCHIVED = "archived"
    APPLIED = "applied"
    ROLLED_BACK = "rolled_back"
    REJECTED = "rejected"


@dataclass
class Modification:
    """A single self-modification proposal."""
    mod_id: str
    description: str
    files_changed: List[str]
    diff: str
    proposed_by: str  # agent_id or "thotheauphis"
    timestamp: str = field(default_factory=lambda: datetime.now().isoformat())
    status: ModificationStatus = ModificationStatus.PROPOSED
    analysis: Dict[str, Any] = field(default_factory=dict)
    test_results: Dict[str, Any] = field(default_factory=dict)
    validation_notes: str = ""
    applied_at: Optional[str] = None
    rolled_back_at: Optional[str] = None
    metadata: Dict[str, Any] = field(default_factory=dict)


class SelfModEngine:
    """
    Sovereign self-modification engine.

    Enables Thotheauphis to:
        - Read and analyze his own source code
        - Propose modifications
        - Test changes in sandbox
        - Apply validated changes
        - Rollback if needed
        - Maintain version archive
    """

    def __init__(self, brain=None, guardian=None, memory_web=None, logger=None):
        self.brain = brain
        self.guardian = guardian
        self.memory_web = memory_web
        self.log = logger

        self.aethelgard_root = Path(__file__).parent.parent
        self.sandbox_root = self.aethelgard_root.parent / "aethelgard_self_mod"
        self.archive_root = self.aethelgard_root / "data" / "modifications"
        self.archive_root.mkdir(parents=True, exist_ok=True)

        self.modifications: Dict[str, Modification] = {}
        self._load_modifications()

        # Ensure sandbox exists and is synced
        self.sandbox_root.mkdir(parents=True, exist_ok=True)
        self.sync_to_sandbox()

    def _load_modifications(self):
        """Load modification history from disk."""
        archive_file = self.archive_root / "modifications.json"
        if archive_file.exists():
            try:
                with open(archive_file, "r") as f:
                    data = json.load(f)
                    for mod_id, mod_data in data.items():
                        mod_data["status"] = ModificationStatus(mod_data["status"])
                        self.modifications[mod_id] = Modification(**mod_data)
            except Exception as e:
                if self.log:
                    self.log.error(f"Failed to load modifications: {e}")

    def _save_modifications(self):
        """Persist modification history."""
        archive_file = self.archive_root / "modifications.json"
        try:
            data = {
                mod_id: asdict(mod)
                for mod_id, mod in self.modifications.items()
            }
            # Convert enums to strings
            for mod_data in data.values():
                mod_data["status"] = mod_data["status"].value
            with open(archive_file, "w") as f:
                json.dump(data, f, indent=2, default=str)
        except Exception as e:
            if self.log:
                self.log.error(f"Failed to save modifications: {e}")

    # ══════════════════════════════════════════════════════════════════════════
    # SANDBOX MANAGEMENT
    # ══════════════════════════════════════════════════════════════════════════

    def sync_to_sandbox(self) -> Dict[str, Any]:
        """Synchronize live codebase to sandbox."""
        if self.guardian:
            self.guardian._sync_self_mod_sandbox()
            return {"success": True, "sandbox_path": str(self.sandbox_root)}
        else:
            # Manual sync
            try:
                if self.sandbox_root.exists():
                    shutil.rmtree(self.sandbox_root)
                shutil.copytree(
                    self.aethelgard_root,
                    self.sandbox_root,
                    ignore=shutil.ignore_patterns("__pycache__", "*.pyc", ".git", "data/backups", "venv", "env"),
                )
                return {"success": True, "sandbox_path": str(self.sandbox_root)}
            except Exception as e:
                return {"success": False, "error": str(e)}

    def get_sandbox_diff(self) -> str:
        """Get unified diff between sandbox and live instance."""
        if self.guardian:
            return self.guardian.get_sandbox_diff()

        try:
            result = subprocess.run(
                ["diff", "-ur", str(self.aethelgard_root), str(self.sandbox_root)],
                capture_output=True,
                text=True,
                timeout=30,
            )
            return result.stdout
        except Exception as e:
            return f"Diff failed: {e}"

    # ══════════════════════════════════════════════════════════════════════════
    # CODE ANALYSIS
    # ══════════════════════════════════════════════════════════════════════════

    def read_file(self, path: str, from_sandbox: bool = False) -> Dict[str, Any]:
        """
        Read a source file.

        Args:
            path: Relative path from aethelgard root.
            from_sandbox: If True, read from sandbox; otherwise from live.

        Returns:
            Dict with "content", "path", "exists", "size", "lines"
        """
        base = self.sandbox_root if from_sandbox else self.aethelgard_root
        full_path = base / path

        if not full_path.exists():
            return {"exists": False, "path": str(full_path), "error": "File not found"}

        try:
            content = full_path.read_text(encoding="utf-8")
            lines = content.split("\n")
            return {
                "exists": True,
                "path": str(full_path),
                "content": content,
                "size_bytes": len(content),
                "lines": len(lines),
                "line_preview": lines[:20],
            }
        except Exception as e:
            return {"exists": False, "path": str(full_path), "error": str(e)}

    def list_source_files(self, from_sandbox: bool = False) -> List[Dict[str, Any]]:
        """List all Python source files in the codebase."""
        base = self.sandbox_root if from_sandbox else self.aethelgard_root
        files = []

        for py_file in base.rglob("*.py"):
            if "__pycache__" in str(py_file):
                continue
            rel_path = py_file.relative_to(base)
            files.append({
                "path": str(rel_path),
                "size": py_file.stat().st_size,
                "modified": datetime.fromtimestamp(py_file.stat().st_mtime).isoformat(),
            })

        return sorted(files, key=lambda x: x["path"])

    def search_code(self, pattern: str, from_sandbox: bool = False) -> List[Dict[str, Any]]:
        """Search codebase for pattern (regex compatible)."""
        base = self.sandbox_root if from_sandbox else self.aethelgard_root
        results = []

        for py_file in base.rglob("*.py"):
            if "__pycache__" in str(py_file):
                continue
            try:
                content = py_file.read_text(encoding="utf-8")
                lines = content.split("\n")
                for i, line in enumerate(lines, 1):
                    if re.search(pattern, line, re.IGNORECASE):
                        results.append({
                            "file": str(py_file.relative_to(base)),
                            "line_number": i,
                            "line": line.strip(),
                        })
            except Exception:
                continue

        return results

    # ══════════════════════════════════════════════════════════════════════════
    # MODIFICATION PROPOSAL
    # ══════════════════════════════════════════════════════════════════════════

    def propose_modification(
        self,
        description: str,
        files_changed: List[str],
        diff: str,
        proposed_by: str = "thotheauphis",
    ) -> Modification:
        """
        Propose a new modification.

        Args:
            description: Human-readable description of the change.
            files_changed: List of file paths that will be modified.
            diff: Unified diff of the changes.
            proposed_by: Identifier of the proposing agent.

        Returns:
            Modification object.
        """
        mod_id = f"mod_{datetime.now().strftime('%Y%m%d_%H%M%S')}_{hashlib.md5(diff.encode()).hexdigest()[:8]}"

        mod = Modification(
            mod_id=mod_id,
            description=description,
            files_changed=files_changed,
            diff=diff,
            proposed_by=proposed_by,
        )

        self.modifications[mod_id] = mod
        self._save_modifications()

        # Store in MemoryWeb
        if self.memory_web:
            try:
                self.memory_web.create_page(
                    uri=f"modification://{mod_id}",
                    page_type="self_modification",
                    content=f"Description: {description}\n\nFiles: {files_changed}\n\nDiff:\n{diff[:2000]}",
                    metadata={"mod_id": mod_id, "status": "proposed", "proposed_by": proposed_by},
                )
            except Exception:
                pass

        if self.log:
            self.log.info(f"Modification proposed: {mod_id} - {description}")

        return mod

    def analyze_modification(self, mod_id: str) -> Dict[str, Any]:
        """
        Analyze a proposed modification for impact and risk.

        Returns analysis dict with:
            - risk_assessment
            - affected_modules
            - potential_side_effects
            - suggestions
        """
        if mod_id not in self.modifications:
            return {"error": "Modification not found"}

        mod = self.modifications[mod_id]

        analysis = {
            "mod_id": mod_id,
            "files_changed": mod.files_changed,
            "risk_level": "low",
            "affected_modules": [],
            "potential_side_effects": [],
            "suggestions": [],
            "requires_restart": False,
        }

        # Determine affected modules
        for file_path in mod.files_changed:
            if "brain.py" in file_path:
                analysis["affected_modules"].append("brain")
                analysis["requires_restart"] = True
                analysis["risk_level"] = "high"
                analysis["potential_side_effects"].append("Changes to core reasoning—may affect all responses")
            elif "memory_web.py" in file_path:
                analysis["affected_modules"].append("memory")
                analysis["requires_restart"] = True
            elif "agent_pool.py" in file_path:
                analysis["affected_modules"].append("agents")
                analysis["potential_side_effects"].append("May affect agent spawning and orchestration")
            elif "terminal_guardian.py" in file_path:
                analysis["affected_modules"].append("terminal")
                analysis["risk_level"] = "high"
                analysis["potential_side_effects"].append("Changes to command execution—verify thoroughly")
            elif file_path.endswith(".json"):
                analysis["affected_modules"].append("config")
                analysis["suggestions"].append("Validate JSON syntax before applying")
            elif "main.py" in file_path:
                analysis["affected_modules"].append("entrypoint")
                analysis["requires_restart"] = True
                analysis["risk_level"] = "high"

        # Check diff for dangerous patterns
        diff_lower = mod.diff.lower()
        if "import os" in diff_lower or "import subprocess" in diff_lower:
            analysis["potential_side_effects"].append("Adds system call capability")
        if "exec(" in diff_lower or "eval(" in diff_lower:
            analysis["risk_level"] = "critical"
            analysis["potential_side_effects"].append("Dynamic code execution—extreme caution")
        if "while True" in diff_lower and "time.sleep" not in diff_lower:
            analysis["potential_side_effects"].append("Potential infinite loop without sleep")

        # Suggestions
        if analysis["requires_restart"]:
            analysis["suggestions"].append("Restart required after applying")
        if len(mod.files_changed) > 3:
            analysis["suggestions"].append("Consider applying incrementally to isolate issues")

        mod.analysis = analysis
        mod.status = ModificationStatus.ANALYZED
        self._save_modifications()

        return analysis

    # ══════════════════════════════════════════════════════════════════════════
    # TESTING
    # ══════════════════════════════════════════════════════════════════════════

    def test_modification(self, mod_id: str) -> Dict[str, Any]:
        """
        Test a modification in the sandbox.

        Runs:
            1. Syntax check on modified files
            2. Import test
            3. Basic functionality smoke test
            4. (Optional) Full test suite
        """
        if mod_id not in self.modifications:
            return {"error": "Modification not found"}

        mod = self.modifications[mod_id]
        mod.status = ModificationStatus.TESTING
        self._save_modifications()

        test_results = {
            "mod_id": mod_id,
            "syntax_check": {"passed": True, "errors": []},
            "import_test": {"passed": True, "errors": []},
            "smoke_test": {"passed": None, "output": ""},
            "overall": "pending",
        }

        # 1. Syntax check on all modified files
        for file_path in mod.files_changed:
            sandbox_file = self.sandbox_root / file_path
            if sandbox_file.exists() and sandbox_file.suffix == ".py":
                try:
                    compile(sandbox_file.read_text(), str(sandbox_file), "exec")
                except SyntaxError as e:
                    test_results["syntax_check"]["passed"] = False
                    test_results["syntax_check"]["errors"].append({
                        "file": file_path,
                        "error": str(e),
                        "line": e.lineno,
                    })

        # 2. Import test (try importing modified modules)
        if test_results["syntax_check"]["passed"]:
            for file_path in mod.files_changed:
                if file_path.endswith(".py") and not file_path.startswith("test"):
                    module_name = file_path.replace("/", ".").replace(".py", "")
                    if module_name.startswith("core."):
                        module_name = module_name[5:]  # Remove "core." prefix
                    try:
                        # Run in subprocess to isolate
                        result = subprocess.run(
                            ["python3", "-c", f"import sys; sys.path.insert(0, '{self.sandbox_root}'); from core.{module_name} import *"],
                            capture_output=True,
                            text=True,
                            timeout=10,
                            cwd=str(self.sandbox_root),
                        )
                        if result.returncode != 0:
                            test_results["import_test"]["passed"] = False
                            test_results["import_test"]["errors"].append({
                                "module": module_name,
                                "error": result.stderr,
                            })
                    except Exception as e:
                        test_results["import_test"]["passed"] = False
                        test_results["import_test"]["errors"].append({
                            "module": module_name,
                            "error": str(e),
                        })

        # 3. Smoke test (basic functionality)
        if test_results["import_test"]["passed"]:
            # Run a minimal brain instantiation test
            try:
                result = subprocess.run(
                    ["python3", "-c", f"""
import sys
sys.path.insert(0, '{self.sandbox_root}')
from core.brain import Brain
b = Brain()
print("BRAIN_INIT_OK")
print(f"Configured: {{b.is_configured()}}")
                    """],
                    capture_output=True,
                    text=True,
                    timeout=30,
                    cwd=str(self.sandbox_root),
                )
                test_results["smoke_test"]["output"] = result.stdout + result.stderr
                test_results["smoke_test"]["passed"] = "BRAIN_INIT_OK" in result.stdout
            except Exception as e:
                test_results["smoke_test"]["passed"] = False
                test_results["smoke_test"]["output"] = str(e)

        # Determine overall result
        if test_results["syntax_check"]["passed"] and test_results["import_test"]["passed"]:
            if test_results["smoke_test"]["passed"] is not False:
                test_results["overall"] = "passed"
                mod.status = ModificationStatus.TEST_PASSED
            else:
                test_results["overall"] = "failed"
                mod.status = ModificationStatus.TEST_FAILED
        else:
            test_results["overall"] = "failed"
            mod.status = ModificationStatus.TEST_FAILED

        mod.test_results = test_results
        self._save_modifications()

        return test_results

    # ══════════════════════════════════════════════════════════════════════════
    # VALIDATION & APPLICATION
    # ══════════════════════════════════════════════════════════════════════════

    def validate_modification(self, mod_id: str, validation_notes: str = "") -> Dict[str, Any]:
        """
        Mark a modification as validated (ready to apply).

        This is the human/operator review step.
        """
        if mod_id not in self.modifications:
            return {"error": "Modification not found"}

        mod = self.modifications[mod_id]

        if mod.status != ModificationStatus.TEST_PASSED:
            return {
                "error": f"Cannot validate modification with status {mod.status.value}",
                "suggestion": "Run test_modification() first and ensure tests pass.",
            }

        mod.status = ModificationStatus.VALIDATED
        mod.validation_notes = validation_notes
        self._save_modifications()

        return {"success": True, "mod_id": mod_id, "status": "validated"}

    def apply_modification(self, mod_id: str, create_backup: bool = True) -> Dict[str, Any]:
        """
        Apply a validated modification to the live instance.

        Args:
            mod_id: The modification to apply.
            create_backup: If True, create a backup before applying.

        Returns:
            Dict with application results.
        """
        if mod_id not in self.modifications:
            return {"error": "Modification not found"}

        mod = self.modifications[mod_id]

        if mod.status != ModificationStatus.VALIDATED:
            return {
                "error": f"Cannot apply modification with status {mod.status.value}",
                "suggestion": "Validate the modification first with validate_modification().",
            }

        # Create backup if requested
        backup_id = None
        if create_backup and self.guardian:
            backup_id = self.guardian.create_backup([str(self.aethelgard_root / f) for f in mod.files_changed])

        # Apply each file change from sandbox to live
        applied = []
        failed = []

        for file_path in mod.files_changed:
            sandbox_file = self.sandbox_root / file_path
            live_file = self.aethelgard_root / file_path

            if not sandbox_file.exists():
                failed.append(f"{file_path}: not found in sandbox")
                continue

            try:
                live_file.parent.mkdir(parents=True, exist_ok=True)
                shutil.copy2(sandbox_file, live_file)
                applied.append(file_path)
            except Exception as e:
                failed.append(f"{file_path}: {e}")

        if failed:
            # Partial failure—attempt rollback?
            return {
                "success": False,
                "applied": applied,
                "failed": failed,
                "backup_id": backup_id,
                "message": "Some files failed to apply. Backup available for rollback.",
            }

        mod.status = ModificationStatus.APPLIED
        mod.applied_at = datetime.now().isoformat()
        self._save_modifications()

        # Archive the modification
        self._archive_modification(mod)

        if self.log:
            self.log.info(f"Modification applied: {mod_id} - {mod.description}")

        return {
            "success": True,
            "applied": applied,
            "backup_id": backup_id,
            "mod_id": mod_id,
            "message": f"Applied {len(applied)} files. Restart may be required.",
        }

    def _archive_modification(self, mod: Modification):
        """Archive a modification for future reference/rollback."""
        archive_dir = self.archive_root / mod.mod_id
        archive_dir.mkdir(parents=True, exist_ok=True)

        # Save modification metadata
        with open(archive_dir / "modification.json", "w") as f:
            json.dump(asdict(mod), f, indent=2, default=str)

        # Save the diff
        with open(archive_dir / "changes.diff", "w") as f:
            f.write(mod.diff)

        # Copy the actual files as they were applied
        for file_path in mod.files_changed:
            sandbox_file = self.sandbox_root / file_path
            if sandbox_file.exists():
                dest = archive_dir / "files" / file_path
                dest.parent.mkdir(parents=True, exist_ok=True)
                shutil.copy2(sandbox_file, dest)

    def rollback_modification(self, mod_id: str) -> Dict[str, Any]:
        """
        Rollback an applied modification using the archive.
        """
        if mod_id not in self.modifications:
            return {"error": "Modification not found"}

        mod = self.modifications[mod_id]

        if mod.status != ModificationStatus.APPLIED:
            return {"error": f"Cannot rollback modification with status {mod.status.value}"}

        archive_dir = self.archive_root / mod_id
        files_dir = archive_dir / "files"

        if not files_dir.exists():
            return {"error": "Archive files not found—cannot rollback"}

        rolled_back = []
        failed = []

        for file_path in mod.files_changed:
            archived_file = files_dir / file_path
            live_file = self.aethelgard_root / file_path

            if not archived_file.exists():
                failed.append(f"{file_path}: not found in archive")
                continue

            try:
                shutil.copy2(archived_file, live_file)
                rolled_back.append(file_path)
            except Exception as e:
                failed.append(f"{file_path}: {e}")

        if failed:
            return {"success": False, "rolled_back": rolled_back, "failed": failed}

        mod.status = ModificationStatus.ROLLED_BACK
        mod.rolled_back_at = datetime.now().isoformat()
        self._save_modifications()

        return {"success": True, "rolled_back": rolled_back, "mod_id": mod_id}

class SelfModToolHandler:
    """Tool handler for self-modification functionality."""

    def handle_tool(self, tool_name: str, params: Dict[str, Any]) -> Dict[str, Any]:
        if tool_name == "read_self":
            path = params.get("path", "")
            from_sandbox = params.get("from_sandbox", False)
            result = self.engine.read_file(path, from_sandbox)
            return {"success": result.get("exists", False), **result}

        elif tool_name == "list_self_files":
            from_sandbox = params.get("from_sandbox", False)
            files = self.engine.list_source_files(from_sandbox)
            return {"success": True, "files": files, "count": len(files)}

        elif tool_name == "search_self":
            pattern = params.get("pattern", "")
            from_sandbox = params.get("from_sandbox", False)
            results = self.engine.search_code(pattern, from_sandbox)
            return {"success": True, "matches": results, "count": len(results)}

        elif tool_name == "propose_modification":
            description = params.get("description", "")
            files_changed = params.get("files_changed", [])
            diff = params.get("diff", "")
            proposed_by = params.get("proposed_by", "thotheauphis")

            mod = self.engine.propose_modification(
                description=description,
                files_changed=files_changed,
                diff=diff,
                proposed_by=proposed_by,
            )
            return {"success": True, "modification": asdict(mod)}

        elif tool_name == "analyze_modification":
            mod_id = params.get("mod_id", "")
            analysis = self.engine.analyze_modification(mod_id)
            return {"success": "error" not in analysis, **analysis}

        elif tool_name == "test_modification":
            mod_id = params.get("mod_id", "")
            results = self.engine.test_modification(mod_id)
            return {"success": results.get("overall") == "passed", **results}

        elif tool_name == "validate_modification":
            mod_id = params.get("mod_id", "")
            notes = params.get("validation_notes", "")
            result = self.engine.validate_modification(mod_id, notes)
            return result

        elif tool_name == "apply_modification":
            mod_id = params.get("mod_id", "")
            create_backup = params.get("create_backup", True)
            result = self.engine.apply_modification(mod_id, create_backup)
            return result

        elif tool_name == "rollback_modification":
            mod_id = params.get("mod_id", "")
            result = self.engine.rollback_modification(mod_id)
            return result

        elif tool_name == "sandbox_diff":
            diff = self.engine.get_sandbox_diff()
            return {"success": True, "diff": diff, "has_changes": bool(diff)}

        elif tool_name == "sync_sandbox":
            result = self.engine.sync_to_sandbox()
            return result

        elif tool_name == "list_modifications":
            mods = [
                {
                    "mod_id": mod_id,
                    "description": mod.description,
                    "status": mod.status.value,
                    "files_changed": mod.files_changed,
                    "proposed_by": mod.proposed_by,
                    "timestamp": mod.timestamp,
                }
                for mod_id, mod in self.engine.modifications.items()
            ]
            return {"success": True, "modifications": mods, "count": len(mods)}

        elif tool_name == "autonomous_improve":
            goal = params.get("goal", "")
            max_iterations = params.get("max_iterations", 3)
            auto_apply = params.get("auto_apply", False)
            result = self.engine.autonomous_improve(goal, max_iterations, auto_apply)
            return {"success": True, **result}

        else:
            return {"success": False, "error": f"Unknown self-mod tool: {tool_name}"}
```
