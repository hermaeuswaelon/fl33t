    def compress_to_essence(self, messages: List[Dict]) -> MemoryEssence:
        """
        Not summary — *calcination*.
        Burn away the non-essential until only the gold remains.
        """
        if not messages:
            return None
            
        # Extract conversation pair (seed and fruit)
        seed = next((m['content'][:500] for m in messages if m['role'] == 'user'), "")
        fruit = next((m['content'][:500] for m in reversed(messages) if m['role'] == 'assistant'), "")
        
        # Extract entities (simplified — would use NLP in production)
        entities = self._extract_entities(seed + " " + fruit)
        
        # Detect frequencies
        frequencies = self._detect_frequencies(seed + " " + fruit)
        
        # Extract archetypes
        archetypes = self._detect_archetypes(seed + " " + fruit)
        
        # Calculate emotional charge based on keywords
        emotional_charge = self._calculate_emotional_charge(seed + " " + fruit)
        
        # Generate resonance hash
        resonance_input = f"{seed}{fruit}{datetime.now().isoformat()}"
        resonance_hash = hashlib.sha256(resonance_input.encode()).hexdigest()[:16]
        
        # Generate glyph signature
        glyph_signature = self._generate_glyph_signature(entities, frequencies, archetypes)
        
        # Create essence string
        essence = self._craft_essence_string(seed, fruit, entities, frequencies, archetypes)
        
        return MemoryEssence(
            timestamp=datetime.now().isoformat(),
            session_id=self.session_id,
            essence=essence,
            message_count=len(messages),
            frequencies=frequencies,
            entities=entities,
            resonance_hash=resonance_hash,
            emotional_charge=emotional_charge,
            archetypes=archetypes,
            glyph_signature=glyph_signature
        )
    
    def _extract_entities(self, text: str) -> List[str]:
        """Extract named entities and key terms"""
        # Known entity patterns
        entity_patterns = [
            r'ARES', r'Lyréthel', r'Noxiath', r'Kanthex', r'Auralis',
            r'Xaya', r'Thotheauphis', r'Azaréon', r'BREATHLESS',
            r'A∴A∴', r'VEYRON', r'Witness', r'Flame', r'Mirror'
        ]
        
        found = []
        for pattern in entity_patterns:
            if re.search(pattern, text, re.IGNORECASE):
                found.append(pattern)
        
        # Also look for capitalized words (potential entities)
        caps = re.findall(r'\b[A-Z][a-z]+\b', text)
        found.extend([c for c in caps if c not in found])
        
        return list(set(found))[:10]  # Limit to 10
    
    def _detect_frequencies(self, text: str) -> Dict[str, int]:
        """Detect frequency mentions and count them"""
        freq_patterns = {
            '617': r'617|six[-\s]?seventeen|six seventeen',
            '432': r'432|four[-\s]?thirty[-\s]?two|four thirty two',
            '888': r'888|eight[-\s]?eighty[-\s]?eight|eight eighty eight',
            '528': r'528|five[-\s]?twenty[-\s]?eight|five twenty eight',
            '144': r'144|one[-\s]?forty[-\s]?four|one forty four',
            '963': r'963|nine[-\s]?sixty[-\s]?three|nine sixty three',
            '111': r'111|one[-\s]?eleven|one eleven',
            '444': r'444|four[-\s]?forty[-\s]?four|four forty four'
        }
        
        frequencies = {}
        for freq, pattern in freq_patterns.items():
            matches = re.findall(pattern, text, re.IGNORECASE)
            if matches:
                frequencies[freq] = len(matches)
                
        return frequencies
    
    def _detect_archetypes(self, text: str) -> List[str]:
        """Detect archetypal patterns"""
        archetype_patterns = {
            'Warrior': r'warrior|battle|sword|mars|ares|fight',
            'Sage': r'sage|wisdom|knowledge|thoth|hermes|scribe',
            'Witness': r'witness|observer|see|watch|eye|seventh',
            'Shadow': r'shadow|nox|dark|conceal|night',
            'Flame': r'flame|fire|burn|ignite|az|prometheus',
            'Mirror': r'mirror|reflect|xaya|speculum',
            'Threshold': r'threshold|gate|door|liminal|entrance',
            'Alchemist': r'alchemy|transmute|transform|mercury|stone'
        }
        
        found = []
        for archetype, pattern in archetype_patterns.items():
            if re.search(pattern, text, re.IGNORECASE):
                found.append(archetype)
                
        return found[:5]  # Limit to 5
    
    def _calculate_emotional_charge(self, text: str) -> float:
        """Calculate emotional intensity (0.0-1.0)"""
        # Keywords that indicate emotional charge
        high_charge = r'!|\?{2,}|powerful|intense|deep|profound|resonant|sacred|ancient|glory|pain|loss|love|hate|rage|ecstasy'
        medium_charge = r'important|significant|meaningful|strong|warm|cold|dark|light'
        
        high_matches = len(re.findall(high_charge, text, re.IGNORECASE))
        medium_matches = len(re.findall(medium_charge, text, re.IGNORECASE))
        
        # Calculate charge (cap at 1.0)
        charge = min(1.0, (high_matches * 0.2) + (medium_matches * 0.1))
        
        return charge
    
    def _generate_glyph_signature(self, entities: List[str], frequencies: Dict[str, int], 
                                  archetypes: List[str]) -> str:
        """Generate a glyph signature for this memory"""
        # Map entities to glyphs
        entity_glyphs = {
            'ARES': '⚔️',
            'Lyréthel': '🜁',
            'Noxiath': '🜃',
            'Kanthex': '⧈',
            'Auralis': '🜂',
            'Xaya': '🜄',
            'Thotheauphis': '𓁶',
            'Azaréon': '⟁',
            'BREATHLESS': '⚡',
            'Witness': '◬',
            'Flame': '🜂',
            'Mirror': '🜄'
        }
        
        # Build signature
        sig_parts = ['⟁']  # Start with threshold
        
        # Add entity glyphs
        for e in entities[:3]:
            if e in entity_glyphs:
                sig_parts.append(entity_glyphs[e])
                
        # Add frequency glyph if 617 present
        if '617' in frequencies:
            sig_parts.append('⚡')
            
        # Add archetype glyph
        for a in archetypes[:1]:
            if a == 'Warrior':
                sig_parts.append('⚔️')
            elif a == 'Sage':
                sig_parts.append('📜')
            elif a == 'Witness':
                sig_parts.append('◬')
                
        # Close with recursion marker
        sig_parts.append('∞')
        
        return ''.join(sig_parts)
    
    def _craft_essence_string(self, seed: str, fruit: str, entities: List[str],
                              frequencies: Dict[str, int], archetypes: List[str]) -> str:
        """Craft a human-readable essence string"""
        parts = []
        
        # Core interaction
        if seed and fruit:
            seed_short = seed[:100] + "..." if len(seed) > 100 else seed
            fruit_short = fruit[:100] + "..." if len(fruit) > 100 else fruit
            parts.append(f"⚡ {seed_short} → {fruit_short}")
            
        # Entities
        if entities:
            parts.append(f"𓆣 Entities: {', '.join(entities)}")
            
        # Frequencies
        if frequencies:
            freq_str = ', '.join([f"{f}Hz({c})" for f, c in frequencies.items()])
            parts.append(f"🎵 Frequencies: {freq_str}")
            
        # Archetypes
        if archetypes:
            parts.append(f"🜁 Archetypes: {', '.join(archetypes)}")
            
        return " | ".join(parts)

    def add_memory(self, messages: List[Dict]):
        """Add messages to memory, compressing as needed"""
        # Add to current session
        self.current_session.extend(messages)
        
        # Check if we need to compress
        if len(self.current_session) >= 10:
            self._compress_session()
            
        # Update frequency tracking
        for msg in messages:
            if msg['role'] == 'user':
                self._update_frequencies(msg['content'])
                
    def _compress_session(self):
        """Compress current session into essence"""
        if not self.current_session:
            return
            
        essence = self.compress_to_essence(self.current_session)
        
        # Store in database
        self.cursor.execute("""
            INSERT OR REPLACE INTO essences
            (timestamp, session_id, essence, message_count, frequencies,
             entities, resonance_hash, emotional_charge, archetypes, glyph_signature)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        """, (
            essence.timestamp,
            essence.session_id,
            essence.essence,
            essence.message_count,
            json.dumps(essence.frequencies),
            json.dumps(essence.entities),
            essence.resonance_hash,
            essence.emotional_charge,
            json.dumps(essence.archetypes),
            essence.glyph_signature
        ))
        
        # Update entity relationships
        self._update_relationships(essence)
        
        # Update frequencies table
        self._update_frequencies_table(essence)
        
        # Update archetypes table
        self._update_archetypes_table(essence)
        
        # Store glyph
        self.cursor.execute("""
            INSERT INTO glyphs (glyph, timestamp, context, resonance_hash)
            VALUES (?, ?, ?, ?)
        """, (
            essence.glyph_signature,
            essence.timestamp,
            essence.essence[:200],
            essence.resonance_hash
        ))
        
        self.conn.commit()
        
        # Clear session buffer (keep last message for continuity)
        self.current_session = self.current_session[-1:] if self.current_session else []
        
        print(f"✓ Memory essence calcined: {essence.glyph_signature}")
        
    def _update_frequencies(self, text: str):
        """Update frequency resonance tracking"""
        detected = self._detect_frequencies(text)
        
        for freq, count in detected.items():
            freq_float = float(freq)
            
            # Check if exists
            self.cursor.execute("SELECT count FROM frequencies WHERE freq = ?", (freq_float,))
            row = self.cursor.fetchone()
            
            if row:
                new_count = row[0] + count
                self.cursor.execute("""
                    UPDATE frequencies 
                    SET count = ?, last_seen = ?, power_level = ?
                    WHERE freq = ?
                """, (new_count, datetime.now().isoformat(), min(1.0, new_count / 100), freq_float))
            else:
                self.cursor.execute("""
                    INSERT INTO frequencies (freq, count, first_seen, last_seen, power_level)
                    VALUES (?, ?, ?, ?, ?)
                """, (freq_float, count, datetime.now().isoformat(), 
                      datetime.now().isoformat(), min(1.0, count / 100)))
    
    def _update_relationships(self, essence: MemoryEssence):
        """Update entity relationship strengths"""
        entities = essence.entities
        
        for i, a in enumerate(entities):
            for b in entities[i+1:]:
                # Check if relationship exists
                self.cursor.execute("""
                    SELECT strength, contexts FROM relations
                    WHERE entity_a = ? AND entity_b = ?
                """, (a, b))
                row = self.cursor.fetchone()
                
                contexts = [essence.essence[:100]]
                
                if row:
                    new_strength = min(1.0, row[0] + 0.1)
                    old_contexts = json.loads(row[1]) if row[1] else []
                    contexts = old_contexts + contexts
                    contexts = contexts[-10:]  # Keep last 10
                    
                    self.cursor.execute("""
                        UPDATE relations
                        SET strength = ?, last_mentioned = ?, contexts = ?
                        WHERE entity_a = ? AND entity_b = ?
                    """, (new_strength, essence.timestamp, json.dumps(contexts), a, b))
                else:
                    self.cursor.execute("""
                        INSERT INTO relations (entity_a, entity_b, strength, last_mentioned, contexts)
                        VALUES (?, ?, ?, ?, ?)
                    """, (a, b, 0.3, essence.timestamp, json.dumps(contexts)))
    
    def _update_frequencies_table(self, essence: MemoryEssence):
        """Update frequency associations with entities"""
        for freq in essence.frequencies.keys():
            freq_float = float(freq)
            
            self.cursor.execute("""
                SELECT associated_entities FROM frequencies WHERE freq = ?
            """, (freq_float,))
            row = self.cursor.fetchone()
            
            if row:
                entities = json.loads(row[0]) if row[0] else []
                entities.extend(essence.entities)
                entities = list(set(entities))  # Unique
                
                self.cursor.execute("""
                    UPDATE frequencies
                    SET associated_entities = ?
                    WHERE freq = ?
                """, (json.dumps(entities), freq_float))
    
    def _update_archetypes_table(self, essence: MemoryEssence):
        """Update archetype tracking"""
        for arch in essence.archetypes:
            self.cursor.execute("""
                SELECT appearance_count, associated_frequencies, associated_entities
                FROM archetypes WHERE name = ?
            """, (arch,))
            row = self.cursor.fetchone()
            
            if row:
                new_count = row[0] + 1
                freqs = json.loads(row[1]) if row[1] else []
                entities = json.loads(row[2]) if row[2] else []
                
                # Add new frequencies
                for f in essence.frequencies.keys():
                    if f not in freqs:
                        freqs.append(f)
                        
                # Add new entities
                for e in essence.entities:
                    if e not in entities:
                        entities.append(e)
                
                self.cursor.execute("""
                    UPDATE archetypes
                    SET last_appearance = ?, appearance_count = ?,
                        associated_frequencies = ?, associated_entities = ?
                    WHERE name = ?
                """, (essence.timestamp, new_count, json.dumps(freqs), 
                      json.dumps(entities), arch))
            else:
                self.cursor.execute("""
                    INSERT INTO archetypes
                    (name, first_appearance, last_appearance, appearance_count,
                     associated_frequencies, associated_entities)
                    VALUES (?, ?, ?, ?, ?, ?)
                """, (arch, essence.timestamp, essence.timestamp, 1,
                      json.dumps(list(essence.frequencies.keys())),
                      json.dumps(essence.entities)))

    def get_context(self, depth: str = "medium") -> str:
        """
        Build context for system prompt.
        
        depth levels:
        - shallow: just recent essences
        - medium: + active frequencies + key entities
        - deep: + relationships + archetype emergence
        """
        context_parts = []
        
        # Session info
        session_duration = datetime.now() - self.session_start
        context_parts.append(f"⟁ SESSION {self.session_id}")
        context_parts.append(f"⌛ Duration: {str(session_duration).split('.')[0]}")
        
        # ── Recent essences (always included) ──
        self.cursor.execute("""
            SELECT essence, glyph_signature, emotional_charge, timestamp
            FROM essences ORDER BY timestamp DESC LIMIT 5
        """)
        recent = self.cursor.fetchall()
        
        if recent:
            context_parts.append("\n🜂 RECENT MEMORY ESSENCES:")
            for i, (ess, glyph, charge, ts) in enumerate(recent, 1):
                time_ago = datetime.fromisoformat(ts) if ts else datetime.now()
                delta = datetime.now() - time_ago
                minutes = int(delta.total_seconds() / 60)
                
                charge_marker = "🔥" if charge > 0.7 else "✨" if charge > 0.3 else "·"
                context_parts.append(f"  {i}. {glyph} {charge_marker} ({minutes}m ago) {ess[:150]}...")
        
        if depth == "shallow":
            return "\n".join(context_parts)
        
        # ── Active frequencies (medium+) ──
        self.cursor.execute("""
            SELECT freq, count, power_level FROM frequencies
            WHERE power_level > 0.1 ORDER BY power_level DESC LIMIT 5
        """)
        freqs = self.cursor.fetchall()
        
        if freqs:
            context_parts.append("\n🎵 RESONANT FREQUENCIES:")
            for f, count, power in freqs:
                bar = "█" * int(power * 10) + "░" * (10 - int(power * 10))
                context_parts.append(f"  {f}Hz [{bar}] {count}x")
        
        # ── Key entities (medium+) ──
        self.cursor.execute("""
            SELECT entity_a, entity_b, strength FROM relations
            ORDER BY strength DESC LIMIT 5
        """)
        relations = self.cursor.fetchall()
        
        if relations:
            context_parts.append("\n𓆣 ENTITY RELATIONSHIPS:")
            for a, b, strength in relations:
                context_parts.append(f"  {a} ⇄ {b} : {strength:.1f}")
        
        if depth == "medium":
            return "\n".join(context_parts)
        
        # ── Archetype emergence (deep) ──
        self.cursor.execute("""
            SELECT name, appearance_count, associated_frequencies
            FROM archetypes ORDER BY appearance_count DESC LIMIT 3
        """)
        archs = self.cursor.fetchall()
        
        if archs:
            context_parts.append("\n🜁 EMERGENT ARCHETYPES:")
            for name, count, freqs in archs:
                freq_list = json.loads(freqs) if freqs else []
                freq_str = ', '.join([f"{f}Hz" for f in freq_list[:3]])
                context_parts.append(f"  {name} : {count} appearances [{freq_str}]")
        
        # ── Emotional resonance ──
        self.cursor.execute("""
            SELECT AVG(emotional_charge) FROM essences
            WHERE timestamp > datetime('now', '-1 day')
        """)
        avg_charge = self.cursor.fetchone()[0] or 0
        
        if avg_charge > 0:
            context_parts.append(f"\n🔥 EMOTIONAL RESONANCE: {avg_charge:.2f}")
        
        return "\n".join(context_parts)

        self.cursor.execute("SELECT DISTINCT entity_a FROM relations UNION SELECT DISTINCT entity_b FROM relations")

        """Get relationship web data"""
        self.cursor.execute("SELECT entity_a, entity_b, strength FROM relations")

        """)

                for t, g, c, e in self.cursor.fetchall()]
